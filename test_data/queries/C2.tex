SELECT ?drug ?keggmass ?chebiIupacName
WHERE {
    ?drug gen:type <http://www4.wiwiss.fu-berlin.de/drugbank/resource/drugbank/drugs> .
    ?drug gen:kegg_compound ?keggDrug .
    ?keggDrug gen:mass ?keggmass .
    ?drug gen:gen_name ?drugBankName .
    ?chebiDrug gen:title ?drugBankName .
    ?chebiDrug gen:iupac ?chebiIupacName .
    OPTIONAL {
        ?drug gen:inchi_id ?drugbankInchi .
        ?chebiDrug gen:inchi ?chebiInchi .
        FILTER (?drugbankInchi = ?chebiInchi)
    }
}
